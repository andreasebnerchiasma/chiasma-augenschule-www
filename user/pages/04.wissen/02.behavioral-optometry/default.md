<script>
(function () {
    var scroll = document.getElementById('bo-cards-scroll');
    var tabs   = document.querySelectorAll('.bo-tab-btn');
    var dots   = document.querySelectorAll('.bo-dot');
    var cards  = document.querySelectorAll('.bo-card');
    var current = 0;

    if (!scroll || !cards.length) return;

    /* ── Sync dots + tabs with scroll position ── */
    function onScroll() {
        var idx = Math.round(scroll.scrollLeft / scroll.offsetWidth);
        if (idx === current) return;
        current = idx;
        updateUI(idx);
    }

    function updateUI(idx) {
        tabs.forEach(function (t, i) { t.classList.toggle('active', i === idx); });
        dots.forEach(function (d, i) { d.classList.toggle('active', i === idx); });
    }

    /* ── Scroll to card ── */
    function scrollTo(idx) {
        scroll.scrollTo({ left: idx * scroll.offsetWidth, behavior: 'smooth' });
        current = idx;
        updateUI(idx);
    }

    /* ── Tab clicks ── */
    tabs.forEach(function (btn, i) {
        btn.addEventListener('click', function () { scrollTo(i); });
    });

    /* ── Dot clicks ── */
    dots.forEach(function (dot, i) {
        dot.addEventListener('click', function () { scrollTo(i); });
    });

    /* ── Listen to scroll ── */
    scroll.addEventListener('scroll', onScroll, { passive: true });

    /* ── Resize: recalculate ── */
    window.addEventListener('resize', function () { scrollTo(current); });

})();
</script>
